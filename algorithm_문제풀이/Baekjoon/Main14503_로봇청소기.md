## Main14503_로봇청소기

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main{
	private static int N;
	private static int M;
	private static int[][] map;
	private static int startX;
	private static int startY;
	private static int startDir;
	static int[] dx = { -1, 0, 1, 0 };
	static int[] dy = { 0, 1, 0, -1 };

	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine(), " ");
		N = Integer.parseInt(st.nextToken());
		M = Integer.parseInt(st.nextToken());
		map = new int[N][M];

		st = new StringTokenizer(br.readLine(), " ");
		startX = Integer.parseInt(st.nextToken());
		startY = Integer.parseInt(st.nextToken());
		startDir = Integer.parseInt(st.nextToken());

		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			for (int j = 0; j < M; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}

		clean(startX, startY, startDir);

	}// end of main

	// Dir 북 동 남 서
	public static void clean(int x, int y, int dir) {
		int result = 0;
		while (true) {

			// 현재 위치를 청소한다.
			if (map[x][y] == 0) {
				result++;
				map[x][y] = 2;
			}

			// 현재 위치에서 현재 방향을 기준으로 왼쪽방향부터 차례대로 탐색을 진행한다.
			// a) 왼쪽 방향에 아직 청소하지 않은 방이 존재한다면, 그 방향으로 회전한 다음 한칸 전진 1번 진행
			int checkDir = dir;
			int count = 0;
			checkDir = (checkDir + 3) % 4;
			int nx = x + dx[checkDir];
			int ny = y + dy[checkDir];
			if (isboolean(nx, ny) && map[nx][ny] == 0) {
				dir = checkDir;
				x = nx;
				y = ny;
				continue;
			} else {
				int tempDir = dir;
				for (int i = 0; i < 4; i++) {
					tempDir = (tempDir + 3) % 4;
					nx = x + dx[tempDir];
					ny = y + dy[tempDir];
					if (isboolean(nx, ny) && map[nx][ny] == 0) {
						dir = tempDir;
						x = nx;
						y = ny;
						break;
					} else {
						count++;
					}

				}
				// 네 방향 모두 청소가 되어있거나 벽인 경우,
				if (count == 4) {
					checkDir = (dir + 2) % 4;
					int bx = x + dx[checkDir];
					int by = y + dy[checkDir];
					if (isboolean(bx, by) && map[bx][by] != 1) {
						x = bx;
						y = by;

					} else {
						break;
					}
				}
			}

		}

		System.out.println(result);
	}

	static boolean isboolean(int x, int y) {
		if (x >= 0 && y >= 0 && x < N && y < M)
			return true;
		return false;

	}
}// end of class
```
